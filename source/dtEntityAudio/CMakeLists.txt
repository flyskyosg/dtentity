SET(LIB_NAME dtEntityAudio)

# add OpenAL include dir
INCLUDE_DIRECTORIES( ${OPENAL_INCLUDE_DIR})

SET(HEADER_PATH ${CMAKE_CURRENT_SOURCE_DIR}/../../include/${LIB_NAME})

SET(LIB_PUBLIC_HEADERS
  ${HEADER_PATH}/export.h
  ${HEADER_PATH}/audiomanager.h
  ${HEADER_PATH}/listener.h
  ${HEADER_PATH}/sound.h
  ${HEADER_PATH}/soundcomponent.h
)

SET(LIB_SOURCES
  audiomanager.cpp
  pluginfactories.cpp
  listener.cpp
  sound.cpp
)

SET(LIB_SOURCES_REPLACE
  soundcomponent.cpp
)

SET(WRAPLIBRARIES 
   ${TARGET_COMMON_LIBRARIES} 
   dtEntityOSG 
   ${OPENAL_LIBRARIES}
   ${OPENAL_UTIL_LIBRARIES}
)
REPLACE_SIDS_IN_SOURCE(LIB_SOURCES_OUT ${LIB_SOURCES_REPLACE})

ADD_LIBRARY(${LIB_NAME} ${DTENTITY_LIBS_DYNAMIC_OR_STATIC}
  ${LIB_PUBLIC_HEADERS}
  ${LIB_SOURCES}
  ${LIB_SOURCES_OUT}
  ${LIB_SOURCES_REPLACE}
)

TARGET_LINK_LIBRARIES( ${LIB_NAME} ${WRAPLIBRARIES})


SET_TARGET_PROPERTIES(${LIB_NAME}
                     PROPERTIES DEFINE_SYMBOL DTENTITY_AUDIO_LIBRARY)
SETUP_LINK_LIBRARIES(${LIB_NAME})
INCLUDE(ModuleInstall OPTIONAL)

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES DEBUG_POSTFIX "${CMAKE_DEBUG_POSTFIX}")
